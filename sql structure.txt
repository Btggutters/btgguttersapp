These are all the postgres query's i used to make my database, so if something needs to be changed give sql query to do so

CREATE TABLE subcontractors (
    id SERIAL PRIMARY KEY,
    company_name VARCHAR(255),
    sales_name VARCHAR(255),
    sales_number VARCHAR(255),
    company_address VARCHAR(255),
    company_price_per_foot NUMERIC
);

CREATE TABLE material (
    id SERIAL PRIMARY KEY,
    size INT CHECK (size = 5 OR size = 6),
    color VARCHAR(255),
    item VARCHAR(255),
    qty INT CHECK (qty >= 0),
    location VARCHAR(255),
    job_id INT,
    FOREIGN KEY (job_id) REFERENCES jobs(id)
);

CREATE TABLE customers (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255),
    email VARCHAR(255),
    phone_number VARCHAR(255)
);


CREATE TABLE jobs (
    id SERIAL PRIMARY KEY,
    drawing_upload BYTEA,
    entity_type VARCHAR(255) CHECK (entity_type IN ('sub', 'customers')),
    customer_id INT,
    subcontractor_id INT,
    address VARCHAR(255),
    job_description VARCHAR(255),
    job_status VARCHAR(255),
    estimate_date DATE,
    estimate_start_time TIME,
    estimate_end_time TIME,
    estimate_duration NUMERIC,
    first_price NUMERIC,
    install_date DATE,
    install_start_time TIME,
    install_end_time TIME,
    install_duration NUMERIC,
    final_price NUMERIC,
    FOREIGN KEY (customer_id) REFERENCES customers(id),
    FOREIGN KEY (subcontractor_id) REFERENCES subcontractors(id),
    CHECK (
        (customer_id IS NOT NULL AND subcontractor_id IS NULL) OR 
        (customer_id IS NULL AND subcontractor_id IS NOT NULL)
    )
);


CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(255),
    password VARCHAR(255)
);
